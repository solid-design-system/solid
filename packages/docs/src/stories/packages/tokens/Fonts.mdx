import { Meta } from '@storybook/addon-docs/blocks';
import theme from 'tailwind-theme';
import cx from 'classix';
import { getTokens } from './utils.js';

<Meta title="Packages/Tokens/Fonts" />

## Font Sizes

<table>
  <tbody>
    {getTokens('text', theme.base, {
      filter: (token) => !token.name.includes('color'),
      sort: (a, b) => {
        const order = new Map(['sm', 'base', 'lg', 'xl', '2xl', '3xl', '4xl'].map((v, i) => [v, i]));
        const aValue = a.name.replace('font-size-', '');
        const bValue = b.name.replace('font-size-', '');

        return order.get(aValue) - order.get(bValue);
      }
    })
      .map((token, i) => {
        return (
          <tr key={token.name}>
            <td>
              <div className="font-bold">sd-font-size-{token.name}</div>
              <div>font-size: {token.value}</div>
            </td>
            <td>
              <div
                className={cx(
                  {
                    sm: '!text-sm',
                    base: '!text-base',
                    lg: '!text-lg',
                    xl: '!text-xl',
                    '2xl': '!text-2xl',
                    '3xl': '!text-3xl',
                    '4xl': '!text-4xl'
                  }[token.name]
                )}
              >
                Lorem ipsum dolor
              </div>
            </td>
          </tr>
        );
      })}

  </tbody>
</table>
