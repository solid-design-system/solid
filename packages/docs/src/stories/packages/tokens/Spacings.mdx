## Spacings

> "By default, these values are inherited by the padding, margin, width, height, maxHeight, flex-basis, gap, inset, space, translate, scrollMargin, scrollPadding, and textIndent core plugins." [tailwindcss docs](https://tailwindcss.com/docs/theme#theme-variable-namespaces)

<table>
  <tbody>
    {theme.base
      .filter(token => token.name.startsWith('spacing'))
      .map(token => ({ ...token, name: token.name.replace('spacing-', '') }))
      .sort((a, b) => {
        const va = a.name === 'spacing' ? '1' : a.name;
        const vb = b.name === 'spacing' ? '1' : b.name;
        return va.localeCompare(vb, undefined, { numeric: true });
      })
      .map((token, i) => {
        const name = token.name === 'spacing' ? '1' : token.name;
        return (
          <tr>
            <td>
              <div class="flex flex-col">
                <div class="font-bold">sd-spacing-{name}</div>
                <div>{token.value}</div>
              </div>
            </td>
            <td class="w-[calc(theme('spacing.40')+3rem)]">
              <div
                class={cx(
                  'bg-primary flex h-10',
                  {
                    '0': 'w-0',
                    '1': 'w-1',
                    '2': 'w-2',
                    '3': 'w-3',
                    '4': 'w-4',
                    '5': 'w-5',
                    '6': 'w-6',
                    '8': 'w-8',
                    '10': 'w-10',
                    '12': 'w-12',
                    '16': 'w-16',
                    '20': 'w-20',
                    '24': 'w-24',
                    '32': 'w-32',
                    '40': 'w-40',
                    '0.25': 'w-0.25',
                    '0.5': 'w-0.5',
                    '1.5': 'w-1.5',
                    auto: 'w-auto'
                  }[name]
                )}
              ></div>
            </td>
          </tr>
        );
      })}
  </tbody>
</table>
